* 这个是文章中的第一类.单纯的找完全相等的数
* 注意两点就是判断条件是<=, mid = low + (high-low) /2=

```C++
int BinarySearch(vector<int> nums, int target) {
	int n = nums.size();
	int low = 0, high = n - 1;
	while (low <= high) {
		int mid = (high - low) / 2 + low;
		if (nums[mid] == target) {
			return mid;
		}

		if (nums[mid] < target)
			low = mid + 1;
		else
			high = mid - 1;
	}

	return -1;
}
```
